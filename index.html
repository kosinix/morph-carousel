Carousel Demo

A no-frills lightweight carousel. Period.

Installation

Include the styles in head

<link rel="stylesheet" href="css/morph-carousel.css">

Include jQuery and the script before the closing body tag:

<script src="js/jquery-1.11.3.min.js"></script>
<script src="js/morph-carousel.min.js"></script>

Prepare the required mark-up:

<div class="morph-carousel">
    <div class="morph-carousel-item">
        <img src="images/1.jpg" alt="image">
    </div>
        ...
    <div class="morph-carousel-item">
        <img src="images/9.jpg" alt="image">
    </div>
</div>

Usage

Initialize the carousel:

<script>
    jQuery( document ).ready(function($) {
        $('#carousel-default').morphCarousel();
    });
</script>

Options

These are the default options:

<script>
    jQuery( document ).ready(function($) {
        $('#carousel').morphCarousel({
            scrollSpeed: 200,
            wrap: false,
            cssClass: {
                carousel: 'morph-carousel',
                inner: 'morph-carousel-inner',
                viewPort: 'morph-carousel-viewport',
                strip: 'morph-carousel-strip',
                item: 'morph-carousel-item',
                prev: 'morph-carousel-prev',
                next: 'morph-carousel-next'
            }
        });
    });
</script>

Enable wrapping:

<script>
    jQuery( document ).ready(function($) {
        $('#carousel-wrap').morphCarousel({
            wrap: true
        });
    });
</script>
